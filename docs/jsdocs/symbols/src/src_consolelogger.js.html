  5 /**
  6  * @namespace logger compatible with console.* calls

  8 var ConsoleLogger	= {};

 11 // export the class in node.js - if running in node.js
 12 if( typeof(window) === 'undefined' )	module.exports	= ConsoleLogger;
 13 
 14 ConsoleLogger.Severity	= {
 15 	'all'		: 0,
 16 	'log'		: 1,
 17 	'warn'		: 2,
 18 	'error'		: 3,
 19 	'nothing'	: 99,
 20 };
 21 
 22 ConsoleLogger.Severity.dfl	= ConsoleLogger.Severity.nothing;
 23 
 24 //////////////////////////////////////////////////////////////////////////////////
 25 //		handle filter								//
 26 //////////////////////////////////////////////////////////////////////////////////
 27 
 28 
 29 ConsoleLogger._filters	= [];
 30 
 31 ConsoleLogger.pushFilter	= function(validFor, severity){
 32 	// sanity check - sanity level MUST be defined
 33 	console.assert(Object.keys(ConsoleLogger.Severity).indexOf(severity) !== -1, 'unknown severity level');
 34 	console.log(validFor instanceof Function);
 35 	// push new level
 36 	ConsoleLogger._filters.push({
 37 		validFor	: validFor,
 38 		severity	: ConsoleLogger.Severity[severity]
 39 	});
 40 };
 41 
 42 ConsoleLogger.filter	= function(severity, stackFrame){
 43 	stackFrame	= stackFrame	|| Stacktrace.parse()[2];
 44 	// sanity check - sanity level MUST be defined
 45 	console.assert(Object.keys(ConsoleLogger.Severity).indexOf(severity) !== -1, 'unknown severity level');
 46 	// go thru all the filters
 47 	var filters	= ConsoleLogger._filters;
 48 	for(var i = 0; i < filters.length; i++){
 49 		var filter	= filters[i];
 50 		if( filter.validFor(stackFrame) === false )	continue;
 51 		return ConsoleLogger.Severity[severity] >= filter.severity ? true : false;
 52 	}
 53 	// if this point is reach, check with Severity.dfl
 54 	return ConsoleLogger.Severity[severity] >= ConsoleLogger.Severity.dfl ? true : false;
 55 }

 61 ConsoleLogger.log	= function(/* ... */){
 62 	if( ConsoleLogger.filter('log') === false )	return;
 63 	console.log.apply(console, arguments);
 64 }

 66 ConsoleLogger.warn	= function(/* ... */){

 68 	console.warn.apply(console, arguments);	
 69 }

 71 ConsoleLogger.error	= function(/* ... */){
 72 	if( ConsoleLogger.filter('error') === false )	return;
 73 	console.error.apply(console, arguments);	
 74 }
