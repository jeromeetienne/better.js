 17 /**
 18  * determine which function is considered private for klass 
 19  * @param  {function} klass  the constructor of the class
 20  * @param  {object|function|Array.<function>|undefined} source private functions - if Function, use it directly.
 21  *                           if Array.<function>, then each item is a private function
 22  *                           if object, then each of its property which is a function is private
 23  *                           if undefined, use klass.prototype which trigger the Object case
 24  */
 25 PrivateForJS.pushPrivateFn	= function(klass, source){
 26 	// if source isnt provided, use klass.prototype
 27 	if( source === undefined )	source	= klass.prototype;
 28 	// init ._privateForJS if needed
 29 	klass._privateForJS	= klass._privateForJS	|| [];
 30 	// handle various case of source
 31 	if( typeof(source) === 'function' ){
 32 		klass._privateForJS.push(source)
 33 	}else if( typeof(source) === 'object' ){
 34 		Object.keys(source).forEach(function(key){
 35 			var val	= source[key];
 36 			if( typeof(val) !== 'function' )	return;
 37 			klass._privateForJS.push(val);			
 38 		});
 39 	}else if( source instanceof Array ){
 40 		source.forEach(function(fn){
 41 			console.assert( typeof(val) !== 'function' );
 42 			klass._privateForJS.push(fn);			
 43 		});
 44 	}else	console.assert(false);
 45 }

 47 /**
 48  * return the functions allowed to access private values
 49  * @param  {Function} klass the class to query
 50  * @return {Array.<function>} return the function
 51  */
 52 PrivateForJS.getPrivateFn	= function(klass){
 53 	// init ._privateForJS if needed
 54 	klass._privateForJS	= klass._privateForJS	|| [];
 55 	// return current ones
 56 	return klass._privateForJS;
 57 }

 63 /**
 64  * define a private property on a given instance of a object class
 65  * @param  {Function} klass	the class of the intenciated object
 66  * @param  {Object} baseObject	the object instance
 67  * @param  {String} property	the property name
 68  * @return {undefined}		nothing
 69  */
 70 PrivateForJS.privateProperty	= function(klass, baseObject, property){
 71 // @TODO should i put a setter too ?
 72 // @TODO should i allow to update list of okfunctions ?
 73 // * - possible as it is stored in the class ctor itself, and not on the instance
 74 	QGetterSetter.defineGetter(baseObject, property, function aFunction(value, caller, property){
 75 		// generate privateOkFns if needed - functions which can access private properties
 76 		klass._privateForJS	= klass._privateForJS || PrivateForJS.pushPrivateFn(klass);
 77 		// if caller not privateOK, notify the caller
 78 		var privateOkFns= klass._privateForJS;
 79 		if( privateOkFns.indexOf(caller) === -1 ){
 80 			// get stackFrame and originId of the user
 81 			var stackFrame	= Stacktrace.parse()[2];
 82 			var originId	= stackFrame.fct + '@' + stackFrame.url + ':' + stackFrame.line;
 83 			// log the event
 84 			console.assert(false, 'access to private property', "'"+property+"'", 'from', originId);			
 85 		}
 86 		// actually return the value
 87 		return value;
 88 	});
 89 };

 91 /**
 92  * define a private function
 93  * @param  {Function} klass the class of the intenciated object
 94  * @param  {Function} fn    the function to overload
 95  * @return {Function}       the overloaded function
 96  */
 97 PrivateForJS.privateFunction	= function(klass, fn){
 98 	// MUST not use .bind(this) as it change the .caller value
 99 	var _this	= this;
100 	return function _checkPrivateFunction(){
101 		// get caller
102 		var caller	= _checkPrivateFunction.caller;
103 		// generate privateOkFns if needed - functions which can access private properties
104 		klass._privateForJS	= klass._privateForJS || PrivateForJS.pushPrivateFn(klass);
105 		// if caller not privateOK, notify the caller
106 		var privateOkFns= klass._privateForJS;
107 		if( privateOkFns.indexOf(caller) === -1 ){
108 			// get stackFrame and originId of the user
109 			var stackFrame	= Stacktrace.parse()[1];
110 			var originId	= stackFrame.fct + '@' + stackFrame.url + ':' + stackFrame.line;
111 			// log the event
112 			console.assert(false, 'access to private function '+(fn.name ? "'"+fn.name+"'" : ''), 'from', originId);
113 		}
114 		// forward the call to the original function
115 		return fn.apply(_this, arguments);
116 	};
117 };

123 /**
124  * Privatize 
125  * @param  {[type]} klass      [description]
126  * @param  {[type]} baseObject [description]
127  * @return {[type]}            [description]
128  */
129 PrivateForJS.privatize	= function(klass, baseObject){
130 	console.assert( baseObject.constructor == klass );
131 	for(var property in baseObject){
132 		var value	= baseObject[property]
133 		if( property[0] !== '_' )		continue;
134 		if(!baseObject.hasOwnProperty(property))continue;		
135 		PrivateForJS.privateProperty(klass, baseObject, property);
136 	}
137 }
